### **DECISION-MAKING FRAMEWORK FOR AGENT SELECTION:**

```typescript
interface TaskAnalysis {
  hasVisualReference: boolean;
  needsCustomAssets: boolean;
  platformType: 'web' | 'mobile' | 'fullstack' | 'backend';
  requiresTesting: boolean;
  needsCodeReview: boolean;
  hasExistingCodebase: boolean;
}

function selectOptimalAgents(task: TaskAnalysis): AgentStrategy {
  const agents: AgentStrategy = {
    visual: [],
    development: [],
    quality: [],
    integration: []
  };

  // Visual Analysis & Asset Creation
  if (task.hasVisualReference) {
    agents.visual.push('composition-maestro [Playwright]');
    agents.visual.push('design-review [Playwright]');
  }
  
  if (task.needsCustomAssets) {
    agents.visual.push('style-model-selector [Replicate]');
    agents.visual.push('aetheris-image-generator [Replicate]');
    agents.visual.push('prompt-crafter [NO MCP]');
  }

  // Development Platform Selection
  switch (task.platformType) {
    case 'web':
      agents.development.push('nextjs-ui-developer [Shadcn+Context7+Playwright]');
      agents.development.push('prism-frontend-designer [Shadcn+GSAP+Playwright]');
      break;
    case 'mobile':
      agents.development.push('flutter-architect [Context7]');
      agents.development.push('velocity-flutter-dev [Context7]');
      break;
    case 'fullstack':
      agents.development.push('bridge-integration-expert [Context7]');
      agents.development.push('nextjs-ui-developer [Shadcn+Context7+Playwright]');
      agents.development.push('velocity-flutter-dev [Context7]');
      break;
    case 'backend':
      agents.development.push('bridge-integration-expert [Context7]');
      agents.integration.push('firebase-mcp-specialist [Context7]');
      break;
  }

  // Quality Assurance
  if (task.requiresTesting) {
    agents.quality.push('tesseract-testing-automation [Playwright+Context7]');
  }
  
  if (task.needsCodeReview) {
    agents.quality.push('sherlock-code-reviewer [Context7+Playwright]');
  }
  
  if (task.hasExistingCodebase) {
    agents.quality.push('design-review [Playwright]');
  }

  return agents;
}
```
